function syncAppointmentFinance(appointment) {
  const existing = state.financeiro.find(f => f.origem === "agendamento" && f.agendamentoId === appointment.id);
  if (appointment.statusPagamento !== "pago") {
    if (existing) {
      state.financeiro = state.financeiro.filter(f => f !== existing);
      appointment.financeiroGerado = false;
    }
    return;
  }
  const payload = {
    tipo: "entrada",
    descricao: `Atendimento - ${appointmentServiceName(appointment)} - ${appointment.nomeCliente}`,
    categoria: "Serviço",
    valor: appointment.valorLiquido,
    data: toDateInput(appointment.dataHoraInicio),
    origem: "agendamento",
    agendamentoId: appointment.id,
  };
  if (existing) {
    Object.assign(existing, payload);
  } else {
    state.financeiro.push({ id: id(), ...payload, dataCadastro: new Date().toISOString() });
    appointment.financeiroGerado = true;
  }
}

function getStatusIcon(appointment) {
  if (appointment.status === "Cancelado") return "⚫";
  const hoje = new Date();
  const dataServico = parseDate(appointment.dataHoraInicio);
  const isAtrasado = (appointment.status === "Agendado" || appointment.status === "Confirmado") && dataServico < hoje;
  if (isAtrasado) return "🔴";
  if (appointment.statusPagamento === "pago") return "🟢";
  if (appointment.status === "Concluído" && appointment.statusPagamento === "pendente") return "🔴";
  if (appointment.status === "Agendado") return "🟡";
  if (appointment.status === "Confirmado") return "🟠";
  return "⚪";
}

function updatePaymentAlert() {
  const hoje = new Date();
  const pendentes = state.agendamentos.filter(a =>
    (a.statusPagamento === "pendente" && a.status !== "Cancelado") ||
    (a.status === "Concluído" && a.statusPagamento === "pendente") ||
    ((a.status === "Agendado" || a.status === "Confirmado") && parseDate(a.dataHoraInicio) < hoje)
  );
  const alertDiv = document.querySelector("#paymentAlertBar");
  if (pendentes.length === 0) {
    alertDiv.style.display = "none";
    return;
  }
  const messages = pendentes.map(a => {
    if (parseDate(a.dataHoraInicio) < hoje && (a.status === "Agendado" || a.status === "Confirmado"))
      return `⏰ ${a.nomeCliente} - atrasado (${new Date(a.dataHoraInicio).toLocaleDateString()})`;
    if (a.statusPagamento === "pendente")
      return `⚠️ ${a.nomeCliente} - pagamento pendente`;
    return `⚠️ ${a.nomeCliente} - pendência`;
  });
  document.querySelector("#alertMessage").innerText = messages.join(" | ");
  alertDiv.style.display = "block";
}
